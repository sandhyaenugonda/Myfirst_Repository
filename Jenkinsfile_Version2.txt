pipeline {
    agent any

    stages {
        stage('Build & Test') {
            steps {
                // Use Maven to run tests
                sh 'mvn -B clean test'
            }
            post {
                always {
                    // Publish JUnit test results to Jenkins
                    junit 'target/surefire-reports/*.xml'
                }
            }
        }

        stage('Package') {
            steps {
                sh 'mvn -B -DskipTests package'
            }
        }

        stage('Archive') {
            steps {
                archiveArtifacts artifacts: 'target/*.jar', fingerprint: true
            }
        }
    }
}
